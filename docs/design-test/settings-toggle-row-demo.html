API ToggleRow — 단일 wrapper + control variant

// 신설 — components/settings/ToggleRow.tsx (또는 ui/primitives.tsx 확장)

type ToggleRowProps = {
  label: string;
  desc?: string;
  tone?: "accent" | "violet";          // active 시 카드 톤
  control:
    | { variant: "switch";  checked: boolean; onChange: (v: boolean) => void; }
    | { variant: "segmented";
        value: string;
        options: Array<{ value: string; label: string }>;
        onChange: (v: string) => void; };
};

// 사용 예 — SettingsDrawer.tsx
<ToggleRow
  label="프롬프트 숨기기 (생성 · 수정 · 영상)"
  desc="ON: 진행 모달 프롬프트 접힘 + ..."
  control={{ variant: "switch", checked: hideAll, onChange: onToggleHideAll }}
/>

<ToggleRow
  label="🧠 AI 보정 모드 기본값"
  desc="페이지 마운트 시점의 기본 모드 (페이지 토글은 session-only · 다음 진입부터 반영)"
  tone="violet"
  control={{
    variant: "segmented",
    value: promptEnhanceMode,
    options: [{ value: "fast", label: "빠른" }, { value: "precise", label: "정밀" }],
    onChange: setPromptEnhanceMode,
  }}
/>
